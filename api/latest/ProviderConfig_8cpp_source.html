15 std::uint64_t libfabricCaps(std::optional<ProviderCapabilities> const& capabilities, bool isTarget)
16 {
17 if (!capabilities)
18 {
19 return 0;
20 }
21 auto result = std::uint64_t{0};
22 if ((capabilities->interfaceCaps & MXL_FABRICS_IFACE_CAP_REMOTE_WRITE) != 0)
23 {
24 result |= (isTarget ? FI_REMOTE_WRITE : FI_WRITE) | FI_RMA;
25 }
26 if ((capabilities->interfaceCaps & MXL_FABRICS_IFACE_CAP_SEND_RECEIVE) != 0)
27 {
28 result |= FI_SEND | FI_RECV;
29 }
30 return result;
31 }

33 std::uint64_t libfabricRequiredCaps(std::optional<ProviderCapabilities> const& capabilities)
34 {
35 if (!capabilities)
36 {
37 return 0;
38 }
39 auto result = std::uint64_t{0};
40 if ((capabilities->interfaceCaps & MXL_FABRICS_IFACE_CAP_REMOTE_WRITE) != 0)
41 {
42 result |= FI_RMA;
43 }
44 return result;
45 }

56 ProviderConfig ProviderConfig::create(Provider provider, bool isTarget, std::optional<ProviderCapabilities> capabilities)
57 {
58 switch (provider)
59 {
60 case Provider::TCP: return ProviderConfig::tcp(isTarget, capabilities);
61 case Provider::VERBS: return ProviderConfig::verbs(isTarget, capabilities);
62 case Provider::EFA: return ProviderConfig::efa(isTarget, capabilities);
63 case Provider::SHM: return ProviderConfig::shm(isTarget, capabilities);
64 default: throw Exception::invalidState("cannot create provider config for ANY provider");
65 }
66 }

128 bool ProviderConfig::isSupportedFabricInfo(FabricInfoView view) const noexcept
129 {
130 // Filters out all protocol types that are not supported for this provider.
131 auto const protocolNotSupported = std::ranges::find(_values.supportedProtocols, view->ep_attr->protocol) == _values.supportedProtocols.end();
132
133 // Filters out all address formats that are not supported for this provider.
134 auto const addressFormatNotSupported =
135 std::ranges::find(_values.supportedAddressFormats, view->addr_format) == _values.supportedAddressFormats.end();
136
137 // Filters out all info objects that have caps set that we can't support.
138 auto const containsFilteredCaps = ((view->caps & _values.filteredCaps) > 0);
139
140 // Filters out all info objects that are missing caps that we always need.
141 auto const missingRequiredCaps = (_values.requiredCaps != 0) && ((view->caps & _values.requiredCaps) != _values.requiredCaps);
142
143 // Filters out all objects that are not the endpoint type that we are looking for with this provider.
144 auto const unsupportedEndpointType = (view->ep_attr->type != _values.endpointType);
145
146 return !(protocolNotSupported || addressFormatNotSupported || containsFilteredCaps || missingRequiredCaps || unsupportedEndpointType);
147 }
