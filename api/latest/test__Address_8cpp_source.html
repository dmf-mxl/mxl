46 OfiSockaddrIb makeSockaddrIb(char const* gid, std::uint16_t pkey, std::uint16_t portSpace, std::uint64_t scopeId, std::uint16_t port)
47 {
48 auto sib = OfiSockaddrIb{};
49 sib.sib_family = AF_IB;
50 sib.sib_pkey = htons(pkey);
51 REQUIRE(::inet_pton(AF_INET6, gid, sib.sib_addr) == 1);
52 sib.sib_sid = htobe64((static_cast<std::uint64_t>(portSpace) << 16) | port);
53 sib.sib_scope_id = htobe64(scopeId);
54 return sib;

98TEST_CASE("ofi: FabricAddress::decode FI_SOCKADDR_IN", "[ofi][FabricAddress]")
99{
100 auto const native = makeSockaddrIn("10.130.40.13", 9000);
101 auto const addr = FabricAddress::decode(FabricAddressFormat::SockaddrIn, &native, sizeof(native));
102
103 REQUIRE_FALSE(addr.empty());
104 REQUIRE(std::holds_alternative<::sockaddr_in>(addr.native()));
105
106 auto const& sin = std::get<::sockaddr_in>(addr.native());
107 REQUIRE(sin.sin_family == AF_INET);
108 REQUIRE(ntohs(sin.sin_port) == 9000);
109
110 char ip[INET_ADDRSTRLEN] = {};
111 REQUIRE(::inet_ntop(AF_INET, &sin.sin_addr, ip, sizeof(ip)) != nullptr);
112 REQUIRE(std::string{ip} == "10.130.40.13");
113
114 // Renders to the canonical ofi_straddr string.
115 REQUIRE(addr.toString() == "fi_sockaddr_in://10.130.40.13:9000");
116}

158TEST_CASE("ofi: FabricAddress::decode FI_SOCKADDR_IB", "[ofi][FabricAddress]")
159{
160 // P_Key, port space and scope id are hexadecimal; the lower 16 bits of the service id carry
161 // the port.
162 auto const native = makeSockaddrIb("fe80::1", /*pkey=*/0x1234, /*portSpace=*/0x10b, /*scopeId=*/0xff, /*port=*/0x1f90);
163 auto const addr = FabricAddress::decode(FabricAddressFormat::SockaddrIb, &native, sizeof(native));
164
165 REQUIRE(std::holds_alternative<OfiSockaddrIb>(addr.native()));
166 auto const& sib = std::get<OfiSockaddrIb>(addr.native());
167 REQUIRE(sib.sib_family == AF_IB);
168 REQUIRE(ntohs(sib.sib_pkey) == 0x1234);
169 REQUIRE(be64toh(sib.sib_scope_id) == 0xff);
170
171 // The service id packs the port space in bits 16..31 and the port in bits 0..15.
172 REQUIRE(((be64toh(sib.sib_sid) >> 16) & 0xfff) == 0x10b);
173 REQUIRE((be64toh(sib.sib_sid) & 0xffff) == 0x1f90);
174
175 // ofi_straddr does not emit the port, so the rendering drops it (but keeps the port space).
176 REQUIRE(addr.toString() == "fi_sockaddr_ib://[fe80::1]:0x1234:0x10b:0xff");
177}
