25 FabricAddress::Native parseInternetAddress(std::string const& ipString, std::uint16_t port = 0)
26 {
27 if (std::ranges::find(ipString, ':') != ipString.end())
28 {
29 auto addr = ::in6_addr{};
30 if (!::inet_pton(AF_INET6, ipString.c_str(), &addr))
31 {
32 throw Exception::invalidArgument("invalid node address");
33 }
34 return FabricAddress::Native{
35 ::sockaddr_in6{.sin6_family = AF_INET6, .sin6_port = port, .sin6_flowinfo = 0, .sin6_addr = addr, .sin6_scope_id = 0}
36 };
37 }
38 else
39 {
40 auto addr = ::in_addr{};
41 if (!::inet_pton(AF_INET, ipString.c_str(), &addr))
42 {
43 throw Exception::invalidArgument("invalid node address");
44 }
45 return FabricAddress::Native{
46 ::sockaddr_in{.sin_family = AF_INET, .sin_port = port, .sin_addr = addr, .sin_zero = {}}
47 };
48 }
49 }

89 std::optional<std::string> toString(OfiSockaddrIb const& sib)
90 {
91 auto const gid = ipToString(AF_INET6, sib.sib_addr);
92 if (!gid)
93 {
94 return std::nullopt;
95 }
96 return fmt::format("fi_sockaddr_ib://[{}]:0x{:x}:0x{:x}:0x{:x}",
97 *gid, // GID
98 ntohs(sib.sib_pkey), // P_Key
99 static_cast<std::uint16_t>((be64toh(sib.sib_sid) >> 16) & 0xfff), // port space
100 static_cast<std::uint8_t>(be64toh(sib.sib_scope_id) & 0xff)); // Scope ID
101 }

189 FabricAddressFormat convertAddressFormat(std::uint32_t fiAddrFormat)
190 {
191 // clang-format off
192 constexpr static auto const valid = std::array<FabricAddressFormat, 8>{
193 FabricAddressFormat::Unspec,
194 FabricAddressFormat::Sockaddr,
195 FabricAddressFormat::SockaddrIp,
196 FabricAddressFormat::SockaddrIn,
197 FabricAddressFormat::SockaddrIn6,
198 FabricAddressFormat::SockaddrIb,
199 FabricAddressFormat::Efa,
200 FabricAddressFormat::String};
201 // clang-format on
202
203 auto val = static_cast<FabricAddressFormat>(fiAddrFormat);
204 if (std::ranges::find(valid, val) == valid.end())
205 {
206 return FabricAddressFormat::Unspec;
207 }
208
209 return val;
210 }

212 FabricAddressFormat mustConvertAddressFormat(std::uint32_t fiAddrFormat)
213 {

216 {
217 throw Exception::invalidArgument("Invalid or unsupported address format value: {}", fiToString(&fiAddrFormat, FI_TYPE_ADDR_FORMAT));
218 }
219
220 return val;
221 }

241 FabricAddress FabricAddress::decode(FabricAddressFormat format, void const* addr, std::size_t len)
242 {
243 // Mirror ofi_straddr: nothing to decode without an address.
244 if (!addr)
245 {
246 return FabricAddress{};
247 }
248
249 auto const sockAddrFamily = [&]
250 {
251 return static_cast<::sockaddr const*>(addr)->sa_family;
252 };
253
254 switch (format)
255 {
256 case FabricAddressFormat::Sockaddr:
257 // Dispatch on the embedded address family (IPv4, IPv6 or IB).
258 switch (sockAddrFamily())
259 {
260 case AF_INET: return decode(FabricAddressFormat::SockaddrIn, addr, len);
261 case AF_INET6: return decode(FabricAddressFormat::SockaddrIn6, addr, len);
262 case AF_IB: return decode(FabricAddressFormat::SockaddrIb, addr, len);
263 default: return FabricAddress{};
264 }
265 case FabricAddressFormat::SockaddrIp:
266 // Like FI_SOCKADDR, but IB is not a valid family here.
267 switch (sockAddrFamily())
268 {
269 case AF_INET: return decode(FabricAddressFormat::SockaddrIn, addr, len);
270 case AF_INET6: return decode(FabricAddressFormat::SockaddrIn6, addr, len);
271 default: return FabricAddress{};
272 }
273 case FabricAddressFormat::SockaddrIn:
274 {
275 if (len < sizeof(::sockaddr_in))
276 {
277 return FabricAddress{};
278 }
279 auto sin = ::sockaddr_in{};
280 std::memcpy(&sin, addr, sizeof(sin));
281 return FabricAddress{sin};
282 }
283 case FabricAddressFormat::SockaddrIn6:
284 {
285 if (len < sizeof(::sockaddr_in6))
286 {
287 return FabricAddress{};
288 }
289 auto sin6 = ::sockaddr_in6{};
290 std::memcpy(&sin6, addr, sizeof(sin6));
291 return FabricAddress{sin6};
292 }
293 case FabricAddressFormat::SockaddrIb:
294 {
295 if (len < sizeof(OfiSockaddrIb))
296 {
297 return FabricAddress{};
298 }
299 auto sib = OfiSockaddrIb{};
300 std::memcpy(&sib, addr, sizeof(sib));
301 return FabricAddress{sib};
302 }
303 case FabricAddressFormat::Efa:
304 {
305 // 16-byte IPv6 GID, 16-bit QPN at offset 16, 32-bit QKey at offset 20.
306 constexpr auto efaAddrLen = std::size_t{24};
307 if (len < efaAddrLen)
308 {
309 return FabricAddress{};
310 }
311 auto const* bytes = static_cast<std::uint8_t const*>(addr);
312 auto efa = EfaAddress{};
313 std::memcpy(efa.gid, bytes, sizeof(efa.gid));
314 std::memcpy(&efa.qpn, bytes + 16, sizeof(efa.qpn));
315 std::memcpy(&efa.qkey, bytes + 20, sizeof(efa.qkey));
316 return FabricAddress{efa};
317 }
318 case FabricAddressFormat::String: return FabricAddress{std::string{static_cast<char const*>(addr)}};
319 default: return FabricAddress{};
320 }
321 }

391 FabricAddress FabricAddress::parse(Provider provider, std::optional<std::string> node, std::optional<std::string> service)
392 {
393 if (!node)
394 {
395 if (service)
396 {
397 throw Exception::invalidArgument("node address is required when service address is specified");
398 }
399
400 return FabricAddress{};
401 }
402
403 switch (provider)
404 {
405 // SHM provider always expected FI_ADDR_STR
406 case Provider::SHM:
407 {
408 if (service && !service->empty())
409 {
410 return FabricAddress{*node + ":" + *service};
411 }
412 else
413 {
414 return FabricAddress{*node};
415 }
416 };
417 // For TCP and Verbs we only support FI_SOCKADDR_IN and FI_SOCKADDR_IN6
418 case Provider::TCP: [[fallthrough]];
419 case Provider::VERBS:
420 {
421 if (service && !service->empty())
422 {
423 auto port = std::stoi(*service);
424 return FabricAddress{parseInternetAddress(*node, port)};
425 }
426 return FabricAddress{parseInternetAddress(*node)};
427 }
428 // EFA only supports FI_ADDR_EFA
429 case Provider::EFA:
430 {
431 auto parsed = parseInternetAddress(*node);
432 auto sin6 = std::get_if<::sockaddr_in6>(&parsed);
433 if (!sin6)
434 {
435 throw Exception::invalidArgument("invalid address format for EFA provider");
436 }
437
438 auto addr = EfaAddress{.gid = {}, .qpn = 0, .qkey = 0};
439 std::memcpy(addr.gid, &sin6->sin6_addr, sizeof(addr.gid));
440 return FabricAddress{addr};
441 }
442 default: throw Exception::invalidState("must specify provider");
443 }
444 }
