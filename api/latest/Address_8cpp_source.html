77 RawFabricAddress RawFabricAddress::retrieveFabricAddress(::fid_t fid, FabricInfoView info)
78 {
79 auto const format = mustConvertAddressFormat(info->addr_format);
80
81 // First obtain the address length
82 std::size_t addrlen = 0;
83 auto ret = fi_getname(fid, nullptr, &addrlen);
84 if (ret != -FI_ETOOSMALL)
85 {
86 throw FabricException("Failed to get address length from endpoint.", MXL_ERR_UNKNOWN, ret);
87 }
88
89 // Now that we have the address length, allocate a receiving buffer and call fi_getname again to retrieve the actual address
90 auto addr = std::vector<std::uint8_t>(addrlen);
91 fiCall(fi_getname, "Failed to retrieve endpoint's local address.", fid, addr.data(), &addrlen);
92
93 return RawFabricAddress{std::move(addr), format};
94 }
