35 [[nodiscard]]
36 bool matchesRequestedAddress(FabricAddress const& requested, FabricInfoView info)
37 {
38 auto provider = providerFromString(info->fabric_attr->prov_name);
39 if (!provider)
40 {
41 return false;
42 }
43
44 if (requested.empty())
45 {
46 return true;
47 }
48
49 if (*provider == Provider::SHM)
50 {
51 return requested.node() == getHostname();
52 }
53
54 try
55 {
56 auto const addressFormat = mustConvertAddressFormat(info->addr_format);
57 auto addr = FabricAddress::decode(addressFormat, info->src_addr, info->src_addrlen);
58 return addr.node() == requested.node();
59 }
60 catch (Exception const& ex)
61 {
62 if (ex.status() != MXL_ERR_INVALID_ARG)
63 {
64 throw;
65 }
66
67 return false;
68 }
69 }

72 FabricInterfaceList probeInterfaces(std::optional<std::reference_wrapper<::mxlFabricsInterfaceConfig const>> query)
73 {
74 // Maps of provider specific configurations and addresses. They are lazily created when they appear on the fabric list.
75 auto providerConfigs = std::map<Provider, ProviderConfig>{};
76 auto fabricAddresses = std::map<Provider, std::optional<FabricAddress>>{};
77 auto getProviderConfig = [&](Provider provider) -> ProviderConfig const&
78 {
79 auto it = providerConfigs.lower_bound(provider);
80 if ((it == providerConfigs.end()) || (it->first != provider))
81 {
82 it = providerConfigs.emplace_hint(it, provider, ProviderConfig::create(provider, false, std::nullopt));
83 }
84
85 return it->second;
86 };
87
88 auto getFabricAddress = [&](Provider provider) -> std::optional<FabricAddress> const&
89 {
90 auto it = fabricAddresses.lower_bound(provider);
91 if ((it == fabricAddresses.end()) || (it->first != provider))
92 {
93 try
94 {
95 it = fabricAddresses.emplace_hint(it,
96 provider,
97 FabricAddress::parse(provider,
98 query ? optStringFromCStr(query.value().get().address.node) : std::nullopt,
99 query ? optStringFromCStr(query.value().get().address.service) : std::nullopt));
100 }
101 catch (Exception const&)
102 {
103 it = fabricAddresses.emplace_hint(it, provider, std::nullopt);
104 }
105 }
106
107 return it->second;
108 };
109
110 // clang-format off
111 auto const requestedProvider = query
112 ? providerFromAPI(query.value().get().provider)
113 : Provider::ANY;
114 // clang-format on
115 if (!requestedProvider)
116 {
117 throw Exception::invalidArgument("Invalid provider");
118 }
119
120 auto list = FabricInterfaceList{};
121 for (auto&& info : FabricInfoList::get())
122 {
123 // Ignore if the provider of this fabric info is not known.
124 auto provider = providerFromString(info->fabric_attr->prov_name);
125 if (!provider)
126 {
127 continue;
128 }
129
130 // Ignore if a specific provider is requested, and this one does not match.
131 if ((requestedProvider != Provider::ANY) && (provider != requestedProvider))
132 {
133 continue;
134 }
135
136 // Ignore if the query address could not be parsed for this provider
137 auto const& address = getFabricAddress(*provider);
138 if (!address)
139 {
140 continue;
141 }
142
143 // Ignore if the info does not match the requested address
144 if (!matchesRequestedAddress(*address, info))
145 {
146 continue;
147 }
148
149 // Ignore if no supported.
150 auto const& config = getProviderConfig(*provider);
151 if (config.isSupportedFabricInfo(info))
152 {
153 auto description = FabricInterfaceDescription::create(info);
154 if (description)
155 {
156 list.push(*description);
157 }
158 }
159 }
160
161 return list;
162 }

164 std::pair<FabricInfo, ProviderConfig> selectSourceInterface(::mxlFabricsInterfaceConfig const& interfaceConfig, bool isTarget)
165 {
166 auto provider = providerFromAPI(interfaceConfig.provider);
167 if (!provider)
168 {
169 throw Exception::invalidArgument("invalid provider");
170 }
171
172 constexpr auto transferCapsMask = std::uint64_t{MXL_FABRICS_IFACE_CAP_REMOTE_WRITE | MXL_FABRICS_IFACE_CAP_SEND_RECEIVE};
173 auto caps = ProviderCapabilities::fromAPI(interfaceConfig.caps);
174 if ((caps.interfaceCaps & transferCapsMask) == 0)
175 {
176 throw Exception::invalidArgument(
177 "Unsupported provider constraints: Missing transfer capability. Need either SEND_RECEIVE or REMOTE_WRITE");
178 }
179 else if ((caps.interfaceCaps & MXL_FABRICS_IFACE_CAP_REMOTE_WRITE) == 0)
180 {
181 throw Exception::noFabric("Unsupported provider constraints: Only REMOTE_WRITE supported at this time.");
182 }
183 if (caps.maxMessageSize == 0)
184 {
185 MXL_WARN("maxMessageSize is not set. This field will be required in a future version.");
186 }
187 auto providerConfig = ProviderConfig::create(*provider, isTarget, caps);
188 auto fabricAddress = FabricAddress::parse(
189 *provider, optStringFromCStr(interfaceConfig.address.node), optStringFromCStr(interfaceConfig.address.service));
190
191 auto sourceInterfaces = FabricInfoList::getSourceInterfaces(providerConfig, fabricAddress);
192 if (std::ranges::distance(sourceInterfaces) == 0)
193 {
194 throw Exception::noFabric("no supported interfaces found");
195 }
196
197 return {*sourceInterfaces.begin(), std::move(providerConfig)};
198 }
