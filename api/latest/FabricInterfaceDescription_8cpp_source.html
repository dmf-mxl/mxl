50 void copyNicProperties(picojson::object& destination, ::fid_nic* nic)
51 {
52 if (nic->device_attr)
53 {
54 auto const device = nic->device_attr;
55 copyStringProperty(destination, device->name, "device_name");
56 copyStringProperty(destination, device->device_id, "device_id");
57 copyStringProperty(destination, device->vendor_id, "device_vendor_id");
58 copyStringProperty(destination, device->device_version, "device_version");
59 copyStringProperty(destination, device->driver, "device_driver");
60 copyStringProperty(destination, device->firmware, "device_firmware");
61 }
62 if (nic->link_attr != nullptr)
63 {
64 auto const link = nic->link_attr;
65 auto linkState = std::string{};
66 switch (link->state)
67 {
68 case FI_LINK_UP: linkState = "up"; break;
69 case FI_LINK_DOWN: linkState = "down"; break;
70 default: linkState = "unknown";
71 }
72 destination["link_state"] = picojson::value{linkState};
73 if (link->speed != 0)
74 {
75 destination["link_speed"] = picojson::value{static_cast<double>(link->speed)};
76 }
77 copyStringProperty(destination, link->address, "link_address");
78 copyStringProperty(destination, link->network_type, "link_type");
79 }
80 if ((nic->bus_attr) && (nic->bus_attr->bus_type == FI_BUS_PCI))
81 {
82 auto const pci = nic->bus_attr->attr.pci;
83 copyNumberProperty(destination, pci.domain_id, "pci_domain_id");
84 copyNumberProperty(destination, pci.bus_id, "pci_bus_id");
85 copyNumberProperty(destination, pci.device_id, "pci_device_id");
86 copyNumberProperty(destination, pci.function_id, "pci_function_id");
87 }
88 }

110 std::optional<FabricInterfaceDescription> FabricInterfaceDescription::create(FabricInfoView info)
111 {
112 auto caps = std::uint64_t{};
113 auto attr = picojson::object{};
114 auto const provider = providerFromString(info->fabric_attr->prov_name);
115 if (!provider)
116 {
117 return std::nullopt;
118 }
119
120 auto faddr = FabricAddress::fromSource(info);
121 auto node = std::string{};
122 auto service = std::string{};
123 if (*provider == Provider::SHM)
124 {
125 node = getHostname();
126 service = faddr.service().value_or("");
127 }
128 else
129 {
130 node = faddr.node().value_or("");
131 }
132
133 if ((info->caps & (FI_SEND | FI_RECV)) != 0)
134 {
135 caps |= MXL_FABRICS_IFACE_CAP_SEND_RECEIVE;
136 }
137 if ((info->caps & (FI_REMOTE_WRITE | FI_WRITE | FI_RMA)) != 0)
138 {
139 caps |= MXL_FABRICS_IFACE_CAP_REMOTE_WRITE;
140 }
141 if ((*provider != Provider::EFA) || (CompletionQueue::isWaitObjectSupportedForEFA()))
142 {
143 caps |= MXL_FABRICS_IFACE_CAP_BLOCKING_OPERATIONS;
144 }
145
146 copyStringProperty(attr, fiToString(&info->addr_format, FI_TYPE_ADDR_FORMAT), "ep_addr_format");
147 copyStringProperty(attr, fiToString(&info->ep_attr->protocol, FI_TYPE_PROTOCOL), "ep_protocol");
148 copyStringProperty(attr, fiToString(&info->ep_attr->type, FI_TYPE_EP_TYPE), "ep_type");
149 copyDomainProperties(attr, info->domain_attr);
150
151 if (info->nic != nullptr)
152 {
153 copyNicProperties(attr, info->nic);
154 }
155
156 if (provider == Provider::TCP)
157 {
158 // For the TCP provider, the device_name is always the domain_attr->name.
159 copyStringProperty(attr, info->domain_attr->name, "device_name");
160 }
161
162 return FabricInterfaceDescription{*provider, std::move(node), std::move(service), caps, info->ep_attr->max_msg_size, std::move(attr)};
163 }
