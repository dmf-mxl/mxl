14 struct InterfaceListStats
15 {
16 int total = 0;
17 int tcp = 0;
18 int shm = 0;
19 };
20
21 [[nodiscard]]
22 InterfaceListStats countInterfaces(mxlFabricsInterfaceList const* list)
23 {
24 auto stats = InterfaceListStats{};
25 for (auto const* entry = list; entry != nullptr; entry = entry->next)
26 {
27 ++stats.total;
28 if (entry->interface.provider == MXL_FABRICS_PROVIDER_TCP)
29 {
30 ++stats.tcp;
31 }
32 else if (entry->interface.provider == MXL_FABRICS_PROVIDER_SHM)
33 {
34 ++stats.shm;
35 }
36 }
37 return stats;
38 }
39}
