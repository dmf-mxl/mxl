175 FabricInfoList FabricInfoList::getSourceInterfaces(ProviderConfig const& providerConfig, std::optional<FabricAddress> const& sourceAddress)
176 {
177 auto info = std::add_pointer_t<::fi_info>{nullptr};
178 auto hints = FabricInfo::empty();
179
180 auto node = std::string{};
181 auto service = std::string{};
182 auto nodeStringBuffer = std::add_pointer_t<char const>{nullptr};
183 auto serviceStringBuffer = std::add_pointer_t<char const>{nullptr};
184
185 // Provider name is owned by the fi_info object and must be ::strdupe'd here.
186 hints->fabric_attr->prov_name = ::strdup(providerConfig.getProviderName().c_str());
187
188 // Set hints defined in the provider config
189 hints->domain_attr->mr_mode = providerConfig.getSupportedMemoryRegistrationModes();
190 hints->ep_attr->type = providerConfig.getEndpointType();
191 hints->caps = providerConfig.getCaps();
192
193 if (sourceAddress)
194 {
195 // If a service name/port exists, create a std::string, and get the c_str()
196 // the std::strings need to live through the call to fi_getinfo, so they are assigned to lvalues
197 // in the parent scope too.
198 if (auto nodeValue = sourceAddress->node(); nodeValue)
199 {
200 node = *nodeValue;
201 nodeStringBuffer = node.c_str();
202 }
203
204 if (auto serviceValue = sourceAddress->service(); serviceValue)
205 {
206 service = *serviceValue;
207 serviceStringBuffer = service.c_str();
208 }
209
210 // The verbs provider need the addr_format hint set for ipv6 to work correctly
211 hints->addr_format = static_cast<std::uint32_t>(sourceAddress->addressFormat());
212 }
213
214 fiCall(::fi_getinfo,
215 "Failed to get source interface information",
216 fiVersion(),
217 nodeStringBuffer,
218 serviceStringBuffer,
219 FI_SOURCE,
220 hints.raw(),
221 &info);
222 return FabricInfoList{info};
223 }
